022package org.openwms.common.location;
023
024import java.util.List;
025
026/**
027 * A LocationService offers some useful methods regarding the general handling of {@link Location}s. <p> This interface is declared generic
028 * typed that implementation classes can use any extension of {@link Location}s. </p>
029 *
030 * @param <T> Any kind of {@link Location}
031 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
032 * @version $Revision$
033 * @since 0.1
034 */
035public interface LocationService<T extends Location> {
036
037    /**
038     * Return a list of all {@link Location}s not sorted and not filtered in natural order.
039     *
040     * @return All {@link Location}s as a list
041     */
042    List<T> getAllLocations();
043
044    /**
045     * Removes a list of {@link Message}s from a Location.
046     *
047     * @param id The technical key of the Location
048     * @param messages The messages to be removed
049     * @return The updated Location
050     */
051    Location removeMessages(Long id, List<Message> messages);
052
053    /**
054     * Return a list of all {@link LocationType}s not sorted and not filtered in natural order.
055     *
056     * @return All {@link LocationType}s as a list
057     */
058    List<LocationType> getAllLocationTypes();
059
060    /**
061     * Create a new {@link LocationType}.
062     *
063     * @param locationType The type to be created
064     */
065    void createLocationType(LocationType locationType);
066
067    /**
068     * Delete already persisted {@link LocationType} instances.
069     *
070     * @param locationTypes A list of all instances to be deleted.
071     */
072    void deleteLocationTypes(List<LocationType> locationTypes);
073
074    /**
075     * Saves a {@link LocationType}.
076     *
077     * @param locationType The type to save
078     * @return The saved type
079     */
080    LocationType saveLocationType(LocationType locationType);
081
082    /**
083     * Find and return a Location identified by the given {@code locationPk}.
084     *
085     * @param locationPK The business key of the Location to search for
086     * @return The Location
087     * @throws org.ameba.exception.NotFoundException if entity not found
088     */
089    Location findByLocationId(LocationPK locationPK);
090}
