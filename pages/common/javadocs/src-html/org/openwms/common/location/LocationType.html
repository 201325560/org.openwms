022package org.openwms.common.location;
023
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.OrderBy;
027import javax.persistence.Table;
028import java.io.Serializable;
029import java.util.Objects;
030
031import org.ameba.integration.jpa.BaseEntity;
032import org.springframework.util.Assert;
033
034/**
035 * A LocationType is the type of {@code Location}s with same characteristics.
036 *
037 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
038 * @version 1.0
039 * @GlossaryTerm
040 * @see Location
041 * @since 0.1
042 */
043@Entity
044@Table(name = "COM_LOCATION_TYPE")
045public class LocationType extends BaseEntity implements Serializable {
046    
047    /** Type of the {@code LocationType} (unique). */
048    @Column(name = "C_TYPE", unique = true, nullable = false)
049    @OrderBy
050    private String type;
051
052    /** Description of the {@code LocationType}. */
053    @Column(name = "C_DESCRIPTION")
054    private String description = DEF_TYPE_DESCRIPTION;
055    /** Default value of the description, by default * {@value} . */
056    public static final String DEF_TYPE_DESCRIPTION = "--";
057
058    /** Length of the {@code LocationType}. */
059    @Column(name = "C_LENGTH")
060    private int length = DEF_LENGTH;
061    /** Default value of {@link #length}. */
062    public static final int DEF_LENGTH = 0;
063
064    /** Width of the {@code LocationType}. */
065    @Column(name = "C_WIDTH")
066    private int width = DEF_WIDTH;
067    /** Default value of {@link #width}. */
068    public static final int DEF_WIDTH = 0;
069
070    /** Height of the {@code LocationType}. */
071    @Column(name = "C_HEIGHT")
072    private int height = DEF_HEIGHT;
073    /** Default value of {@link #height}. */
074    public static final int DEF_HEIGHT = 0;
075
076    /*~ ----------------------------- constructors ------------------- */
077    /**
078     * Dear JPA...
079     */
080    protected LocationType() {
081    }
082
083    /**
084     * Create a new {@code LocationType} with an unique natural key.
085     *
086     * @param type Unique type
087     */
088    public LocationType(String type) {
089        Assert.hasText(type, "type must exist when creating a new LocationType");
090        this.type = type;
091    }
092
093    /* ----------------------------- methods ------------------- */
094    /**
095     * Returns the unique identifier of the {@code LocationType}.
096     *
097     * @return type The Type
098     */
099    public String getType() {
100        return this.type;
101    }
102
103    /**
104     * Returns the length of the {@code LocationType}.
105     *
106     * @return length The Length
107     */
108    public int getLength() {
109        return this.length;
110    }
111
112    /**
113     * Set the length of this {@code LocationType}.
114     *
115     * @param length The length of this type
116     */
117    public void setLength(int length) {
118        this.length = length;
119    }
120
121    /**
122     * Returns the width of this {@code LocationType}.
123     *
124     * @return width The Width
125     */
126    public int getWidth() {
127        return this.width;
128    }
129
130    /**
131     * Set the width of this {@code LocationType}.
132     *
133     * @param width The width of this type
134     */
135    public void setWidth(int width) {
136        this.width = width;
137    }
138
139    /**
140     * Returns the height of the {@code LocationType}.
141     *
142     * @return height The Height
143     */
144    public int getHeight() {
145        return this.height;
146    }
147
148    /**
149     * Set the height of this {@code LocationType}.
150     *
151     * @param height The height of this type
152     */
153    public void setHeight(int height) {
154        this.height = height;
155    }
156
157    /**
158     * Returns the description of this {@code LocationType}.
159     *
160     * @return description The description text
161     */
162    public String getDescription() {
163        return this.description;
164    }
165
166    /**
167     * Set the description of the {@code LocationType}.
168     *
169     * @param description The description text of this type
170     */
171    public void setDescription(String description) {
172        this.description = description;
173    }
174
175    /**
176     * Return the type as String.
177     *
178     * @return String
179     * @see java.lang.Object#toString()
180     */
181    @Override
182    public String toString() {
183        return getType();
184    }
185
186    /**
187     * {@inheritDoc}
188     */
189    @Override
190    public boolean equals(Object o) {
191        if (this == o) return true;
192        if (o == null || getClass() != o.getClass()) return false;
193        LocationType that = (LocationType) o;
194        return Objects.equals(type, that.type);
195    }
196
197    /**
198     * {@inheritDoc}
199     */
200    @Override
201    public int hashCode() {
202        return Objects.hash(type);
203    }
204}
