022package org.openwms.common.transport;
023
024import javax.persistence.Column;
025import javax.persistence.Embeddable;
026import java.io.Serializable;
027
028import org.apache.commons.lang3.StringUtils;
029
030
031/**
032 * A Barcode is a printable item with an unique identifier to label {@code TransportUnit}s. The identifier has a defined number of
033 * characters whereas these characters are aligned either left or right. Non filled positions of a Barcode are padded with a so called
034 * padding character.
035 *
036 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
037 * @version 1.0
038 * @GlossaryTerm
039 * @since 0.1
040 */
041@Embeddable
042public class Barcode implements Serializable {
043
044    /** Length of a Barcode field. */
045    public static final int BARCODE_LENGTH = 20;
046
047    /**
048     * A BARCODE_ALIGN defines whether the {@code Barcode} is applied {@code LEFT} or {@code RIGHT}. <p> Only be used when padding is
049     * activated. </p>
050     *
051     * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
052     * @version 1.0
053     * @since 0.1
054     */
055    public static enum BARCODE_ALIGN {
056        /** Barcode is left aligned. */
057        LEFT,
058        /** Barcode is right aligned. */
059        RIGHT
060    }
061
062    /** Define whether to use character padding or not. */
063    private static boolean padded = true;
064
065    /**
066     * Defines a character used for padding.<br> If the actually length of the {@code Barcode} is less than the maximum defined {@code
067     * length} the rest will be filled with {@code padder} characters.
068     */
069    private static char padder = '0';
070
071    /** Defines the maximum length of characters. */
072    private static int length = Barcode.BARCODE_LENGTH;
073
074    /** The alignment of the {@code Barcode}. Could be something of {@link BARCODE_ALIGN}. */
075    private static BARCODE_ALIGN alignment = BARCODE_ALIGN.RIGHT;
076
077    /** 'Identifier' of the {@code Barcode}. <p> <i>Note:</i>It is not guaranteed that this field must be unique. </p> */
078    @Column(name = "C_BARCODE")
079    private String value;
080
081    /*~ ----------------------------- constructors ------------------- */
082
083    /** Dear JPA... */
084    protected Barcode() {
085    }
086
087    /**
088     * Create a new {@code Barcode} with a String.
089     *
090     * @param value The value of the {@code Barcode} as String
091     * @throws IllegalArgumentException when the value is set to <code>null</code>.
092     */
093    public Barcode(String value) {
094        adjustBarcode(value);
095    }
096
097    /*~ ----------------------------- methods ------------------- */
098
099    /**
100     * Force the Barcode to be aligned to the determined rules regarding padding, alignment.
101     *
102     * @param val The old Barcode as String
103     * @return The new aligned Barcode
104     */
105    public final String adjustBarcode(String val) {
106        if (val == null) {
107            throw new IllegalArgumentException("Cannot create a barcode without value");
108        }
109        if (isPadded()) {
110            this.value = (alignment == BARCODE_ALIGN.RIGHT) ? StringUtils.leftPad(val, length, padder) : StringUtils
111                    .rightPad(val, length, padder);
112        } else {
113            this.value = val;
114        }
115        return this.value;
116    }
117
118    /**
119     * Returns the alignment.
120     *
121     * @return The alignment
122     */
123    public static BARCODE_ALIGN getAlignment() {
124        return alignment;
125    }
126
127    /**
128     * Set the alignment.
129     *
130     * @param align The alignment to set
131     */
132    public static void setAlignment(BARCODE_ALIGN align) {
133        alignment = align;
134    }
135
136    /**
137     * Check if {@code Barcode} is padded.
138     *
139     * @return {@literal true} if {@code Barcode} is padded, otherwise {@literal false}.
140     */
141    public static boolean isPadded() {
142        return padded;
143    }
144
145    /**
146     * Set padded.
147     *
148     * @param p {@literal true} if {@code Barcode} should be padded, otherwise {@literal false}.
149     */
150    static void setPadded(boolean p) {
151        padded = p;
152    }
153
154    /**
155     * Return the padding character.
156     *
157     * @return The padding character.
158     */
159    public static char getPadder() {
160        return padder;
161    }
162
163    /**
164     * Set the padding character.
165     *
166     * @param p The padding character to use
167     */
168    static void setPadder(char p) {
169        padder = p;
170        padded = true;
171    }
172
173    /**
174     * Return the {@code Barcode} value.
175     *
176     * @return The value of the {@code Barcode}
177     */
178    public String getValue() {
179        return value;
180    }
181
182    /**
183     * Set the {@code Barcode} value.
184     *
185     * @param value The value to set
186     */
187    public void setValue(String value) {
188        adjustBarcode(value);
189    }
190
191    /**
192     * Return the length.
193     *
194     * @return The length
195     */
196    public static int getLength() {
197        return length;
198    }
199
200    /**
201     * Set the length.
202     *
203     * @param l The length to set
204     */
205    static void setLength(int l) {
206        length = l;
207    }
208
209    /**
210     * Return the value of the {@code Barcode} as String.
211     *
212     * @return As String
213     * @see java.lang.Object#toString()
214     */
215    @Override
216    public String toString() {
217        return value;
218    }
219
220    /**
221     * {@inheritDoc}
222     */
223    @Override
224    public int hashCode() {
225        final int prime = 31;
226        int result = super.hashCode();
227        result = prime * result + ((value == null) ? 0 : value.hashCode());
228        return result;
229    }
230
231    /**
232     * {@inheritDoc}
233     */
234    @Override
235    public boolean equals(Object obj) {
236        if (this == obj) {
237            return true;
238        }
239        if (obj == null) {
240            return false;
241        }
242        if (getClass() != obj.getClass()) {
243            return false;
244        }
245        Barcode other = (Barcode) obj;
246        if (value == null) {
247            if (other.value != null) {
248                return false;
249            }
250        } else if (!value.equals(other.value)) {
251            return false;
252        }
253        return true;
254    }
255}
