053/**
054 * A TransportUnit is an item like a box, a toad, a bin or a palette that is moved around within a warehouse and can carry goods. <p> Used
055 * as container to transport items like {@code LoadUnit}s. It can be moved between {@code Location}s. </p>
056 *
057 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
058 * @version 1.0
059 * @GlossaryTerm
060 * @since 0.1
061 */
062@Entity
063@Table(name = "COM_TRANSPORT_UNIT", uniqueConstraints = @UniqueConstraint(columnNames = {"C_BARCODE"}))
064public class TransportUnit extends ApplicationEntity {
065
066    /** Unique natural key. */
067    @Embedded
068    @AttributeOverride(name = "value", column = @Column(name = "C_BARCODE", length = Barcode.BARCODE_LENGTH))
069    @OrderBy
070    private Barcode barcode;
071
072    /** Indicates whether the {@code TransportUnit} is empty or not (nullable). */
073    @Column(name = "C_EMPTY")
074    private Boolean empty;
075
076    /** Date when the {@code TransportUnit} has been moved to the current {@link Location}. */
077    @Temporal(TemporalType.TIMESTAMP)
078    @Column(name = "C_ACTUAL_LOCATION_DATE")
079    private Date actualLocationDate;
080
081    /** Weight of the {@code TransportUnit}. */
082    @Embedded
083    @AttributeOverride(name = "quantity", column = @Column(name = "C_WEIGHT", length = CoreTypeDefinitions.QUANTITY_LENGTH))
084    private Weight weight = new Weight("0");
085
086    /** State of the {@code TransportUnit}. */
087    @Column(name = "C_STATE")
088    @Enumerated(EnumType.STRING)
089    private TransportUnitState state = TransportUnitState.AVAILABLE;
090
091    /** The current {@link Location} of the {@code TransportUnit}. */
092    @ManyToOne
093    @JoinColumn(name = "C_ACTUAL_LOCATION", nullable = false)
094    private Location actualLocation;
095
096    /** The target {@link Location} of the {@code TransportUnit}.<br /> This property will be set when a {@code TransportOrder} is started. */
097    @ManyToOne
098    @JoinColumn(name = "C_TARGET_LOCATION")
099    private Location targetLocation;
100
101    /** The {@link TransportUnitType} of the {@code TransportUnit}. */
102    @ManyToOne
103    @JoinColumn(name = "C_TRANSPORT_UNIT_TYPE", nullable = false)
104    private TransportUnitType transportUnitType;
105
106    /** Owning {@code TransportUnit}. */
107    @ManyToOne
108    @JoinColumn(name = "C_PARENT")
109    private TransportUnit parent;
110
111    /** The {@code User} who performed the last inventory action on the {@code TransportUnit}. */
112    @Column(name = "C_INVENTORY_USER")
113    private String inventoryUser;
114
115    /** Date of last inventory check. */
116    @Temporal(TemporalType.TIMESTAMP)
117    @Column(name = "C_INVENTORY_DATE")
118    private Date inventoryDate;
119
120    /** A set of all child {@code TransportUnit}s, ordered by id. */
121    @OneToMany(mappedBy = "parent", cascade = {CascadeType.MERGE, CascadeType.PERSIST})
122    @OrderBy("id DESC")
123    private Set<TransportUnit> children = new HashSet<>();
124
125    /** A Map of errors occurred on the {@code TransportUnit}. */
126    @OneToMany(cascade = CascadeType.ALL)
127    @JoinTable(name = "COM_TRANSPORT_UNIT_ERROR", joinColumns = @JoinColumn(name = "C_TRANSPORT_UNIT_ID"), inverseJoinColumns = @JoinColumn(name = "C_ERROR_ID"))
128    // TODO [openwms]: 12/07/16 refactor into a JPA2 map withou using Date as key!
129    private Map<Date, UnitError> errors = new HashMap<>();
130
131    /*~ ----------------------------- constructors ------------------- */
132
133    /** Dear JPA... */
134    protected TransportUnit() {
135    }
136
137    /**
138     * Create a new {@code TransportUnit} with an unique id. The id is used to create a {@link Barcode}.
139     *
140     * @param unitId The unique identifier of the {@code TransportUnit}
141     */
142    TransportUnit(String unitId) {
143        Assert.hasText(unitId);
144        this.barcode = new Barcode(unitId);
145    }
146
147    /**
148     * Create a new {@code TransportUnit} with an unique {@link Barcode}.
149     *
150     * @param barcode The unique identifier of this {@code TransportUnit} is the {@link Barcode}
151     */
152    TransportUnit(Barcode barcode) {
153        this.barcode = new Barcode(barcode.adjustBarcode(barcode.getValue()));
154    }

393    /**
394     * Add a {@code TransportUnit} to the children.
395     *
396     * @param transportUnit The {@code TransportUnit} to be added to the list of children
397     * @throws IllegalArgumentException when transportUnit is {@literal null}
398     */
399    public void addChild(TransportUnit transportUnit) {
400        Assert.notNull(transportUnit, "Child to add may not be null, this: " + this);
401        if (transportUnit.hasParent()) {
402            if (transportUnit.getParent().equals(this)) {
403
404                // if this instance is already the parent, we just return
405                return;
406            }
407
408            // disconnect post from it's current relationship
409            transportUnit.getParent().removeChild(transportUnit);
410        }
411
412        // make this instance the new parent
413        transportUnit.setParent(this);
414        children.add(transportUnit);
415    }
416
417    boolean hasParent() {
418        return parent != null;
419    }
420
421    /**
422     * Remove a {@code TransportUnit} from the list of children.
423     *
424     * @param transportUnit The {@code TransportUnit} to be removed from the list of children
425     * @throws IllegalArgumentException when transportUnit is {@literal null} or any other failure occurs
426     */
427    public void removeChild(TransportUnit transportUnit) {
428        Assert.notNull(transportUnit, "Child to remove may not be null, this: " + this);
429        // make sure we are the parent before we break the relationship
430        if (transportUnit.parent != null && transportUnit.getParent().equals(this)) {
431            transportUnit.setParent(null);
432            children.remove(transportUnit);
433        } else {
434            throw new IllegalArgumentException("Child transportUnit not associated with this instance");
435        }
436    }

458    /**
459     * {@inheritDoc}
460     * <p>
461     * Uses barcode for calculation.
462     */
463    @Override
464    public int hashCode() {
465        final int prime = 31;
466        int result = super.hashCode();
467        result = prime * result + ((barcode == null) ? 0 : barcode.hashCode());
468        return result;
469    }
470
471    /**
472     * {@inheritDoc}
473     * <p>
474     * Uses barcode for comparison.
475     */
476    @Override
477    public boolean equals(Object obj) {
478        if (this == obj) {
479            return true;
480        }
481        if (obj == null) {
482            return false;
483        }
484        if (!(obj instanceof TransportUnit)) {
485            return false;
486        }
487        TransportUnit other = (TransportUnit) obj;
488        if (barcode == null) {
489            if (other.barcode != null) {
490                return false;
491            }
492        } else if (!barcode.equals(other.getBarcode())) {
493            return false;
494        }
495        return true;
496    }
