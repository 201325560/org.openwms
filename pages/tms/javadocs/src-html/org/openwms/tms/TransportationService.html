022package org.openwms.tms;
023
024import java.util.Collection;
025
026/**
027 * A TransportationService offers some useful methods regarding the general handling of {@link TransportOrder}s.
028 *
029 * @param <T> Any kind of {@link TransportOrder}
030 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
031 * @version 1.1
032 * @since 0.1
033 */
034public interface TransportationService<T extends TransportOrder> {
035
036    /**
037     * Returns the number of {@code TransportOrder}s that have the {@code target} as target and are in one of the {@code states}.
038     *
039     * @param target The target place to search TransportOrders for
040     * @param states An array of TransportOrder states to filter TransportOrders for
041     * @return Number of all TransportOrders in one of the {@code states} that are on the way to the {@code target}
042     */
043    int getNoTransportOrdersToTarget(String target, String... states);
044
045    /**
046     * Create a new {@link TransportOrder} with the given {@code target}.
047     *
048     * @param barcode The ID of the {@code TransportUnit} to move
049     * @param target The target of the TransportOrder to move to
050     * @param priority A {@link PriorityLevel} of the new {@link TransportOrder}
051     * @return The newly created {@link TransportOrder}
052     */
053    T create(String barcode, String target, PriorityLevel priority);
054
055    /**
056     * Modifies an existing {@link TransportOrder} according to the argument passed as {@code transportOrder}.
057     *
058     * @param transportOrder Stores the ID of the {@link TransportOrder} to change as well as the state to change
059     * @return The modified instance
060     */
061    T update(T transportOrder);
062
063    /**
064     * Try to turn a list of {@link TransportOrder}s into the given {@code state}.
065     *
066     * @param pKeys The persisted keys of {@link TransportOrder}s
067     * @param state The state to change all orders to
068     * @return A list of persisted keys of {@link TransportOrder}s that have not been changed
069     * @deprecated Use update instead
070     */
071    @Deprecated
072    Collection<String> change(Collection<String> pKeys, TransportOrder.State state);
073
074    /**
075     * Find and return the {@code TransportOrder} identified by the persisted key {@code pKey}.
076     *
077     * @param pKey The persisted key
078     * @return An {@code TransportOrder} instance
079     * @throws org.ameba.exception.NotFoundException if no entity was found
080     */
081    T findByPKey(String pKey);
082}
