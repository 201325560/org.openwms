186    private void validateInitializationCondition() {
187        if (transportUnitBK == null || transportUnitBK.isEmpty() || (targetLocation == null && targetLocationGroup == null)) {
188            throw new StateChangeException(String.format("Not all properties set to turn TransportOrder into next state! transportUnit's barcode [%s], targetLocation [%s], targetLocationGroup [%s]", transportUnitBK, targetLocation, targetLocationGroup));
189        }
190    }
191
192    /**
193     * Validate whether a state change is valid or not. States must be changed in a defined order. Mostly the order is defined by the
194     * ordering if the states in {@link TransportOrder.State} enum class. But some other rules are checked here too and an exception is
195     * thrown in case the sequence of states is violated.
196     *
197     * @param newState The new state of the order
198     * @throws StateChangeException when <li>newState is {@literal null} or</li><li>the state shall be turned back to a prior state
199     * or</li><li>when the caller tries to leap the state {@link TransportOrder.State#INITIALIZED}</li>
200     */
201    private void validateStateChange(TransportOrder.State newState) throws StateChangeException {
202        if (newState == null) {
203            throw new StateChangeException(translator.translate(TMSMessageCodes.TO_STATE_CHANGE_NULL_STATE), TMSMessageCodes.TO_STATE_CHANGE_NULL_STATE, getPersistentKey());
204        }
205        if (state.compareTo(newState) > 0) {
206            // Don't allow to turn back the state!
207            throw new StateChangeException(translator.translate(TMSMessageCodes.TO_STATE_CHANGE_BACKWARDS_NOT_ALLOWED, getPersistentKey()), TMSMessageCodes.TO_STATE_CHANGE_BACKWARDS_NOT_ALLOWED, getPersistentKey());
208        }
209        switch (state) {
210            case CREATED:
211                if (newState != INITIALIZED && newState != CANCELED) {
212                    throw new StateChangeException(translator.translate(TMSMessageCodes.TO_STATE_CHANGE_NOT_READY, newState, getPersistentKey()), TMSMessageCodes.TO_STATE_CHANGE_NOT_READY, newState, getPersistentKey());
213                }
214                validateInitializationCondition();
215                break;
216            case INITIALIZED:
217                if (newState != STARTED && newState != CANCELED && newState != ONFAILURE) {
218                    throw new StateChangeException(translator.translate(TMSMessageCodes.STATE_CHANGE_ERROR_FOR_INITIALIZED_TO, getPersistentKey()), TMSMessageCodes.STATE_CHANGE_ERROR_FOR_INITIALIZED_TO, getPersistentKey());
219                }
220                if (newState == STARTED && startedTOExists()) {
221                    throw new StateChangeException(translator.translate(TMSMessageCodes.START_TO_NOT_ALLOWED_ALREADY_STARTED_ONE, transportUnitBK, getPersistentKey()), TMSMessageCodes.START_TO_NOT_ALLOWED_ALREADY_STARTED_ONE, transportUnitBK, getPersistentKey());
222                }
223                break;
224            case STARTED:
225                // new state may be one of the following, no additional if-check required here
226                break;
227            case FINISHED:
228            case ONFAILURE:
229            case CANCELED:
230                throw new StateChangeException("Not allowed to change the state of a TransportOrder that has already been completed. Current state is " + state);
231            default:
232                throw new IllegalStateException("TO state not managed:" + state);
233        }
234    }
235
236    private boolean startedTOExists() {
237        return repo.numberOfTransportOrders(transportUnitBK, STARTED) > 0;
238    }
239
240    /**
241     * Change the state of the {@code TransportOrder} regarding some rules.
242     *
243     * @param newState The new state of the order
244     * @throws StateChangeException in case <ul> <li>the newState is {@literal null} or</li> <li>the newState is less than the old state
245     * or</li> <li>the {@code TransportOrder} is in state {@link TransportOrder.State#CREATED} and shall be manually turned into something
246     * else then {@link TransportOrder.State#INITIALIZED} or {@link TransportOrder.State#CANCELED}</li> <li>the {@code TransportOrder} is
247     * {@link TransportOrder.State#CREATED} and shall be {@link TransportOrder.State#INITIALIZED} but it is incomplete</li> </ul>
248     */
249    public void setState(TransportOrder.State newState) throws StateChangeException {
250        validateStateChange(newState);
251        switch (newState) {
252            case STARTED:
253                startDate = new Date();
254                break;
255            case FINISHED:
256            case ONFAILURE:
257            case CANCELED:
258                endDate = new Date();
259                break;
260            default:
261                // OK for all others
262        }
263        state = newState;
264        // TODO [openwms]: 24/07/16 publish state changed here!
265    }

357    /**
358     * Check whether one of the targets has changed between this TransportOrder and the one passed as {@code transportOrder}.
359     *
360     * @param transportOrder The TransportOrder to verify against
361     * @return {@literal true} if targets has changed, otherwise {@literal false}
362     */
363    boolean hasTargetChanged(TransportOrder transportOrder) {
364        return ((targetLocation != null && !targetLocation.equals(transportOrder.getTargetLocation())) ||
365                (targetLocationGroup != null && targetLocationGroup.equals(transportOrder.getTargetLocationGroup())));
366    }
367
368    public static enum State {
369        /** Status of new created {@code TransportOrder}s. */
370        CREATED(10),
371
372        /** Status of a full initialized {@code TransportOrder}, ready to be started. */
373        INITIALIZED(20),
374
375        /** A started and active{@code TransportOrder}, ready to be executed. */
376        STARTED(30),
377
378        /** Status to indicate that the {@code TransportOrder} is paused. Not active anymore. */
379        INTERRUPTED(40),
380
381        /** Status to indicate a failure on the {@code TransportOrder}. Not active anymore. */
382        ONFAILURE(50),
383
384        /** Status of a aborted {@code TransportOrder}. Not active anymore. */
385        CANCELED(60),
386
387        /** Status to indicate that the {@code TransportOrder} completed successfully. */
388        FINISHED(70);
389
390        private final int order;
391
392        private State(int sortOrder) {
393            this.order = sortOrder;
394        }
395
396        /**
397         * Get the order.
398         *
399         * @return the order.
400         */
401        public int getOrder() {
402            return order;
403        }
404    }
