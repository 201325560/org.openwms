022package org.openwms.tms;
023
024/**
025 * A TMSMessageCodes.
026 *
027 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
028 * @since 1.0
029 */
030public class TMSMessageCodes {
031
032    /** Signals that the TransportOrder with the given persisted key wasn't found. */
033    public static final String TO_WITH_PKEY_NOT_FOUND = "TMS.TO_WITH_PKEY_NOT_FOUND";
034    /** Signals an exception because it was tried to turn back a TransportOrder into a state that isn't allowed. */
035    public static final String TO_STATE_CHANGE_BACKWARDS_NOT_ALLOWED = "TMS.TO_STATE_CHANGE_BACKWARDS_NOT_ALLOWED";
036    /** Signals an exception because it was tried to change the state of a TransportOrder into a following but not allowed state. */
037    public static final String TO_STATE_CHANGE_NOT_READY = "TMS.TO_STATE_CHANGE_NOT_READY";
038    /** Signals that a request with state of NULL. */
039    public static final String TO_STATE_CHANGE_NULL_STATE = "TMS.TO_STATE_CHANGE_NULL_STATE";
040    /** Signals an exception that it is not allowed to start a TransportOrder, because there is already a started one. */
041    public static final String START_TO_NOT_ALLOWED_ALREADY_STARTED_ONE = "TMS.START_TO_NOT_ALLOWED_ALREADY_STARTED_ONE";
042    /** Signals an exception that the requested state change is not allowed for the initialized TransportOrder. */
043    public static final String STATE_CHANGE_ERROR_FOR_INITIALIZED_TO="TMS.STATE_CHANGE_ERROR_FOR_INITIALIZED_TO";
044}
