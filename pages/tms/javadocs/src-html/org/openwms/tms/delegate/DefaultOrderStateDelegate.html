022package org.openwms.tms.delegate;
023
024import java.util.Collections;
025import java.util.List;
026
027import org.ameba.exception.NotFoundException;
028import org.openwms.common.CommonGateway;
029import org.openwms.tms.TransportOrder;
030import org.openwms.tms.TransportOrderRepository;
031import org.openwms.tms.TransportStartComparator;
032import org.openwms.tms.exception.StateChangeException;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035import org.springframework.beans.factory.annotation.Autowired;
036import org.springframework.context.annotation.Lazy;
037import org.springframework.stereotype.Component;
038import org.springframework.transaction.annotation.Propagation;
039import org.springframework.transaction.annotation.Transactional;
040
041/**
042 * A DefaultOrderStateDelegate. Lazy instantiated, only when needed. Thus it is possible to override this bean and prevent instantiation.
043 *
044 * @author <a href="mailto:russelltina@users.sourceforge.net">Tina Russell</a>
045 * @version $Revision$
046 * @since 0.1
047 */
048@Transactional(propagation = Propagation.MANDATORY)
049@Lazy
050@Component
051public class DefaultOrderStateDelegate implements TransportOrderStateDelegate {
052
053    private static final Logger LOGGER = LoggerFactory.getLogger(DefaultOrderStateDelegate.class);
054    private TransportOrderRepository dao;
055    private TransportOrderStarter starter;
056    private CommonGateway commonGateway;
057
058    /**
059     * Create a new DefaultOrderStateDelegate.
060     *
061     * @param dao TransportOrderDao is required
062     * @param starter TransportOrderStarter is required
063     * @param commonGateway The common gateway
064     */
065    @Autowired
066    public DefaultOrderStateDelegate(TransportOrderRepository dao, TransportOrderStarter starter, CommonGateway commonGateway) {
067        this.dao = dao;
068        this.starter = starter;
069        this.commonGateway = commonGateway;
070    }
071
072    /**
073     * {@inheritDoc}
074     * <p>
075     * Search for already {@link TransportOrder.State#CREATED} {@link TransportOrder}s for this transportUnit and try to initialize them.
076     * When initialization is done try to start them.
077     */
078    @Override
079    public void afterCreation(Long pk) {
080        TransportOrder to = dao.findOne(pk);
081        List<TransportOrder> transportOrders = findInState(to.getTransportUnitBK(), TransportOrder.State.CREATED);
082        for (TransportOrder transportOrder : transportOrders) {
083            boolean go = initialize(transportOrder);
084            if (go) {
085                try {
086                    starter.start(transportOrder);
087                } catch (StateChangeException sce) {
088                    // Not starting a transport here is not a problem, so be
089                    // quiet
090                    LOGGER.warn(sce.getMessage());
091                }
092            }
093        }
094    }
095
096    /**
097     * {@inheritDoc}
098     * <p>
099     * Just search the next TransportOrder for the TransportUnit and try to start it.
100     */
101    @Override
102    public void afterFinish(Long id) {
103        startNextForTu(id);
104    }
105
106    /**
107     * {@inheritDoc}
108     * <p>
109     * Just search the next TransportOrder for the TransportUnit and try to start it.
110     */
111    @Override
112    public void onCancel(Long id) {
113        startNextForTu(id);
114    }
115
116    /**
117     * {@inheritDoc}
118     * <p>
119     * Just search the next TransportOrder for the TransportUnit and try to start it.
120     */
121    @Override
122    public void onFailure(Long id) {
123        startNextForTu(id);
124    }
125
126    private void startNextForTu(Long id) {
127        TransportOrder transportOrder = dao.findOne(id);
128        if (null == transportOrder) {
129            LOGGER.warn("TransportOrder with id:" + id + " could not be loaded");
130            return;
131        }
132        List<TransportOrder> transportOrders = findInState(transportOrder.getTransportUnitBK(),
133                TransportOrder.State.INITIALIZED);
134        Collections.sort(transportOrders, new TransportStartComparator());
135        for (TransportOrder to : transportOrders) {
136            try {
137                starter.start(to);
138                break;
139            } catch (StateChangeException sce) {
140                if (LOGGER.isWarnEnabled()) {
141                    // Not starting a transport here is not a problem, so be
142                    // quiet
143                    LOGGER.warn(sce.getMessage(), sce);
144                }
145            }
146        }
147    }
148
149    /**
150     * {@inheritDoc}
151     * <p>
152     * Just search the next TransportOrder for the TransportUnit and try to start it.
153     */
154    @Override
155    public void onInterrupt(Long id) {
156        startNextForTu(id);
157    }
158
159    private boolean initialize(TransportOrder transportOrder) {
160        try {
161            transportOrder.setState(TransportOrder.State.INITIALIZED);
162        } catch (StateChangeException sce) {
163            LOGGER.info("Could not initialize TransportOrder [" + transportOrder.getPk() + "]. Message:"
164                    + sce.getMessage());
165            return false;
166        }
167        transportOrder.setSourceLocation(commonGateway.getTransportUnit(transportOrder.getTransportUnitBK()).orElseThrow(NotFoundException::new).getActualLocation().toString());
168        if (LOGGER.isDebugEnabled()) {
169            LOGGER.debug("TransportOrder " + transportOrder.getPk() + " INITIALIZED");
170        }
171        return true;
172    }
173
174    private List<TransportOrder> findInState(String transportUnitBK, TransportOrder.State... orderStates) {
175        return dao.findByTransportUnitBKAndStates(transportUnitBK, orderStates);
176    }
177}
