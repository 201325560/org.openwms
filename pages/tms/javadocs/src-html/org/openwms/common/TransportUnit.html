022package org.openwms.common;
023
024import java.io.Serializable;
025import java.util.Objects;
026
027import org.openwms.tms.targets.Location;
028
029/**
030 * A TransportUnit.
031 *
032 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
033 * @version 1.0
034 * @since 1.0
035 */
036public class TransportUnit implements Serializable {
037
038    private String barcode;
039    private Location actualLocation;
040    private String target;
041
042    public TransportUnit() {
043
044    }
045
046    public TransportUnit(String barcode, Location actualLocation, String target) {
047        this.barcode = barcode;
048        this.actualLocation = actualLocation;
049        this.target = target;
050    }
051
052    public String getBarcode() {
053        return barcode;
054    }
055
056    public void setBarcode(String barcode) {
057        this.barcode = barcode;
058    }
059
060    public Location getActualLocation() {
061        return actualLocation;
062    }
063
064    public void setActualLocation(Location actualLocation) {
065        this.actualLocation = actualLocation;
066    }
067
068    public String getTarget() {
069        return target;
070    }
071
072    public void setTarget(String target) {
073        this.target = target;
074    }
075
076    public void clearTarget() {
077        this.target = "";
078    }
079
080    @Override
081    public boolean equals(Object o) {
082        if (this == o) return true;
083        if (o == null || getClass() != o.getClass()) return false;
084        TransportUnit that = (TransportUnit) o;
085        return Objects.equals(barcode, that.barcode) &&
086                Objects.equals(actualLocation, that.actualLocation) &&
087                Objects.equals(target, that.target);
088    }
089
090    @Override
091    public int hashCode() {
092        return Objects.hash(barcode, actualLocation, target);
093    }
094}
