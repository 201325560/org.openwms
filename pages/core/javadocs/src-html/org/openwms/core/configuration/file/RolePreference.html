022package org.openwms.core.configuration.file;
023
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.EnumType;
027import javax.persistence.Enumerated;
028import javax.persistence.NamedQueries;
029import javax.persistence.NamedQuery;
030import javax.persistence.Table;
031import javax.persistence.UniqueConstraint;
032import javax.xml.bind.annotation.XmlAttribute;
033import javax.xml.bind.annotation.XmlTransient;
034import javax.xml.bind.annotation.XmlType;
035import java.io.Serializable;
036
037import org.openwms.core.configuration.PropertyScope;
038import org.springframework.util.Assert;
039
040/**
041 * A RolePreference is used to provide settings specific to an {@code Role} . These kind of {@link Preferences} is valid for the assigned
042 * Role only. {@code User}s assigned to a {@code Role} inherit these RolePreferences but a RolePreference can be overruled by an {@link
043 * UserPreference}. RolePreferences can be defined within a preferences file but also be created with the UI.
044 *
045 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
046 * @version $Revision$
047 * @GlossaryTerm
048 * @since 0.1
049 */
050@XmlType(name = "rolePreference", namespace = "http://www.openwms.org/schema/usermanagement")
051@Entity
052@Table(name = "COR_ROLE_PREFERENCE", uniqueConstraints = @UniqueConstraint(columnNames = {"C_TYPE", "C_OWNER", "C_KEY"}))
053@NamedQueries({
054        @NamedQuery(name = RolePreference.NQ_FIND_BY_OWNER, query = "select rp from RolePreference rp where rp.owner = :owner")})
055public class RolePreference extends AbstractPreference implements Serializable {
056
057    /**
058     * Type of this preference.
059     */
060    @XmlTransient
061    @Enumerated(EnumType.STRING)
062    @Column(name = "C_TYPE")
063    private PropertyScope type = PropertyScope.ROLE;
064    /**
065     * Owner of the {@code RolePreference}.
066     */
067    @XmlAttribute(name = "owner", required = true)
068    @Column(name = "C_OWNER")
069    private String owner;
070    /**
071     * Key value of the {@link RolePreference}.
072     */
073    @XmlAttribute(name = "key", required = true)
074    @Column(name = "C_KEY")
075    private String key;
076
077    /**
078     * Query to find <strong>all</strong> {@code RolePreference}s of a {@code Role}. <li>Query parameter name <strong>owner</strong> : The
079     * rolename of the {@code Role} to search for.</li><br /> Name is {@value} .
080     */
081    public static final String NQ_FIND_BY_OWNER = "RolePreference" + FIND_BY_OWNER;
082
083    /**
084     * Create a new RolePreference. Defined for the JAXB implementation.
085     */
086    public RolePreference() {
087        super();
088    }
089
090    /**
091     * Create a new RolePreference.
092     *
093     * @param rolename The name of the Role that owns this preference
094     * @param key the key
095     * @throws IllegalArgumentException when rolename or key is {@literal null} or empty
096     */
097    public RolePreference(String rolename, String key) {
098        // Called from the client.
099        super();
100        Assert.hasText(owner, "Not allowed to create a RolePreference with an empty rolename");
101        Assert.hasText(key, "Not allowed to create a RolePreference with an empty key");
102        owner = rolename;
103        this.key = key;
104    }
105
106    /**
107     * Get the key.
108     *
109     * @return the key.
110     */
111    public String getKey() {
112        return key;
113    }
114
115    /**
116     * Get the name of the {@code Role} as String.
117     *
118     * @return the rolename.
119     */
120    public String getOwner() {
121        return owner;
122    }
123
124    /**
125     * {@inheritDoc}
126     *
127     * @see AbstractPreference#getType()
128     */
129    @Override
130    public PropertyScope getType() {
131        return PropertyScope.ROLE;
132    }
133
134    /**
135     * {@inheritDoc}
136     *
137     * @see AbstractPreference#getFields()
138     */
139    @Override
140    protected Object[] getFields() {
141        return new Object[]{getType(), getOwner(), getKey()};
142    }
143
144    /**
145     * {@inheritDoc}
146     * <p>
147     * Uses the type, owner and the key to create a {@link PreferenceKey} instance.
148     *
149     * @see AbstractPreference#getPrefKey()
150     */
151    @Override
152    public PreferenceKey getPrefKey() {
153        return new PreferenceKey(getType(), getOwner(), getKey());
154    }
155
156    /**
157     * {@inheritDoc}
158     * <p>
159     * Uses key, owner and type for hashCode calculation.
160     *
161     * @see java.lang.Object#hashCode()
162     */
163    @Override
164    public int hashCode() {
165        final int prime = 31;
166        int result = super.hashCode();
167        result = prime * result + ((key == null) ? 0 : key.hashCode());
168        result = prime * result + ((owner == null) ? 0 : owner.hashCode());
169        result = prime * result + ((type == null) ? 0 : type.hashCode());
170        return result;
171    }
172
173    /**
174     * {@inheritDoc}
175     * <p>
176     * Comparison done with key, owner and type fields. Not delegated to super class.
177     *
178     * @see java.lang.Object#equals(java.lang.Object)
179     */
180    @Override
181    public boolean equals(Object obj) {
182        if (null == obj) {
183            return false;
184        }
185        if (this == obj) {
186            return true;
187        }
188        if (getClass() != obj.getClass()) {
189            return false;
190        }
191        RolePreference other = (RolePreference) obj;
192        if (key == null) {
193            if (other.key != null) {
194                return false;
195            }
196        } else if (!key.equals(other.key)) {
197            return false;
198        }
199        if (owner == null) {
200            if (other.owner != null) {
201                return false;
202            }
203        } else if (!owner.equals(other.owner)) {
204            return false;
205        }
206        return type == other.type;
207    }
208
209    /**
210     * {@inheritDoc}
211     * <p>
212     * Use all fields.
213     */
214    @Override
215    public String toString() {
216        return "RolePreference{" +
217                "type=" + type +
218                ", owner='" + owner + '\'' +
219                ", key='" + key + '\'' +
220                "} " + super.toString();
221    }
222}
