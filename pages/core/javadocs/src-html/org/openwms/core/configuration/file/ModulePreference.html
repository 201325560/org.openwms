022package org.openwms.core.configuration.file;
023
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.EnumType;
027import javax.persistence.Enumerated;
028import javax.persistence.NamedQueries;
029import javax.persistence.NamedQuery;
030import javax.persistence.Table;
031import javax.persistence.UniqueConstraint;
032import javax.xml.bind.annotation.XmlAttribute;
033import javax.xml.bind.annotation.XmlTransient;
034import javax.xml.bind.annotation.XmlType;
035import java.io.Serializable;
036
037import org.openwms.core.configuration.PropertyScope;
038import org.springframework.util.Assert;
039
040/**
041 * A ModulePreference is used to store configuration settings in Module scope. <p> The table model of an ModulePreference spans an unique
042 * key over the columns C_TYPE, C_OWNER and C_KEY. </p> <p> It's counterpart in the context of JAXB is the modulePreference element. </p>
043 *
044 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
045 * @version 0.2
046 * @GlossaryTerm
047 * @since 0.1
048 */
049@XmlType(name = "modulePreference", namespace = "http://www.openwms.org/schema/preferences")
050@Entity
051@Table(name = "COR_MODULE_PREFERENCE", uniqueConstraints = @UniqueConstraint(columnNames = {"C_TYPE", "C_OWNER",
052        "C_KEY"}))
053@NamedQueries({
054        @NamedQuery(name = ModulePreference.NQ_FIND_BY_OWNER, query = "select mp from ModulePreference mp where mp.owner = :owner")})
055public class ModulePreference extends AbstractPreference implements Serializable {
056
057    /**
058     * Query to find <strong>all</strong> {@code ModulePreference}s of a {@code Module}. <li>Query parameter name <strong>owner</strong> :
059     * The modulename of the {@code Module} to search for.</li><br /> Name is {@value} .
060     */
061    public static final String NQ_FIND_BY_OWNER = "ModulePreference" + FIND_BY_OWNER;
062
063    /**
064     * Type of this preference.
065     */
066    @XmlTransient
067    @Enumerated(EnumType.STRING)
068    @Column(name = "C_TYPE")
069    private PropertyScope type = PropertyScope.MODULE;
070
071    /**
072     * Owner of the {@code ModulePreference} (not nullable).
073     */
074    @XmlAttribute(name = "owner", required = true)
075    @Column(name = "C_OWNER", nullable = false)
076    private String owner;
077
078    /**
079     * Key of the {@code ModulePreference} (not nullable).
080     */
081    @XmlAttribute(name = "key", required = true)
082    @Column(name = "C_KEY", nullable = false)
083    private String key;
084
085    /**
086     * Create a new {@code ModulePreference}. Only defined by the JAXB implementation.
087     */
088    public ModulePreference() {
089        super();
090    }
091
092    /**
093     * Create a new {@code ModulePreference}.
094     *
095     * @param owner The name of the owning module
096     * @param key the key
097     * @throws IllegalArgumentException when key or owner is {@literal null} or empty
098     */
099    public ModulePreference(String owner, String key) {
100        // Called from the client-side only.
101        super();
102        Assert.hasText(owner, "Not allowed to create an ModulePreference with an empty owner");
103        Assert.hasText(key, "Not allowed to create an ModulePreference with an empty key");
104        this.owner = owner;
105        this.key = key;
106    }
107
108    /**
109     * Get the key.
110     *
111     * @return the key
112     */
113    public String getKey() {
114        return key;
115    }
116
117    /**
118     * Get the owner.
119     *
120     * @return the owner
121     */
122    public String getOwner() {
123        return owner;
124    }
125
126    /**
127     * {@inheritDoc}
128     */
129    @Override
130    public PropertyScope getType() {
131        return type;
132    }
133
134    /**
135     * {@inheritDoc}
136     */
137    @Override
138    protected Object[] getFields() {
139        return new Object[]{getType(), getOwner(), getKey()};
140    }
141
142    /**
143     * {@inheritDoc}
144     * <p>
145     * Uses the type, owner and the key to create a {@link PreferenceKey} instance.
146     */
147    @Override
148    public PreferenceKey getPrefKey() {
149        return new PreferenceKey(getType(), getOwner(), getKey());
150    }
151
152    /**
153     * {@inheritDoc}
154     * <p>
155     * Uses the type, owner and the key for the hashCode calculation.
156     */
157    @Override
158    public int hashCode() {
159        final int prime = 31;
160        int result = super.hashCode();
161        result = prime * result + ((key == null) ? 0 : key.hashCode());
162        result = prime * result + ((owner == null) ? 0 : owner.hashCode());
163        result = prime * result + ((type == null) ? 0 : type.hashCode());
164        return result;
165    }
166
167    /**
168     * {@inheritDoc}
169     * <p>
170     * Comparison done with the type, owner and the key fields. Not delegated to super class.
171     */
172    @Override
173    public boolean equals(Object obj) {
174        if (null == obj) {
175            return false;
176        }
177        if (this == obj) {
178            return true;
179        }
180        if (getClass() != obj.getClass()) {
181            return false;
182        }
183        ModulePreference other = (ModulePreference) obj;
184        if (key == null) {
185            if (other.key != null) {
186                return false;
187            }
188        } else if (!key.equals(other.key)) {
189            return false;
190        }
191        if (owner == null) {
192            if (other.owner != null) {
193                return false;
194            }
195        } else if (!owner.equals(other.owner)) {
196            return false;
197        }
198        return type == other.type;
199    }
200
201    /**
202     * {@inheritDoc}
203     * <p>
204     * Use all fields.
205     */
206    @Override
207    public String toString() {
208        return "ModulePreference{" +
209                "type=" + type +
210                ", owner='" + owner + '\'' +
211                ", key='" + key + '\'' +
212                "} " + super.toString();
213    }
214}
