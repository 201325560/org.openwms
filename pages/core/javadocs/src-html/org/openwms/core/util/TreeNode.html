022package org.openwms.core.util;
023
024import java.util.Iterator;
025import java.util.Map;
026
027/**
028 * A TreeNode to define a tree data structure.
029 * 
030 * @param <T>
031 *            The type of node.
032 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
033 * @version $Revision$
034 * @since 0.1
035 */
036public interface TreeNode<T> {
037
038    /**
039     * Get the data of the node.
040     * 
041     * @return The attached Node data
042     */
043    T getData();
044
045    /**
046     * Set the data of the node.
047     * 
048     * @param data
049     *            Data to set
050     */
051    void setData(T data);
052
053    /**
054     * Return whether this node is leaf.
055     * 
056     * @return <code>true</code> if this node is leaf, otherwise <code>false</code>
057     */
058    boolean isLeaf();
059
060    /**
061     * Get all child entries. Each child entry contains an identifier (key) and a child node (value).
062     * 
063     * Identifiers are used to generate model keys representing paths to tree nodes. Model keys are used:
064     * <ol>
065     * <li>For persisting tree state. That means identifiers should be serializable when some JSF features are used, e.g. client-side state
066     * saving.</li>
067     * <li>For construction of client identifiers. Client identifier for tree nodes consists of String representations of identifiers
068     * separated by a separator char. A String representation of identifier should be a valid XML ID, e.g. conform to <a
069     * href="http://www.w3.org/TR/xml11/#NT-Name"> XML Name Production</a> production.</li>
070     * </ol>
071     * 
072     * @return {@link Iterator} over Map.Entry instances containing {@link TreeNode} as values and their identifiers as keys
073     * 
074     */
075    Iterator<Map.Entry<Object, TreeNode<T>>> getChildren();
076
077    /**
078     * Find a child by id.
079     * <p>
080     * See {@link #getChildren()} for more information about the identifier constraints.
081     * </p>
082     * 
083     * @param id
084     *            The identifier of the child to find
085     * @return designated {@link TreeNode} instance or <code>null</code>
086     */
087    TreeNode<T> getChild(Object id);
088
089    /**
090     * Adds a child to the children.
091     * <p>
092     * See {@link #getChildren()} for more information about identifier constraints.
093     * </p>
094     * 
095     * @param identifier
096     *            child identifier
097     * @param child
098     *            The child
099     */
100    void addChild(Object identifier, TreeNode<T> child);
101
102    /**
103     * Removes a child from the children collection.
104     * <p>
105     * See {@link #getChildren()} for more information about identifier constraints.
106     * </p>
107     * 
108     * @param id
109     *            The id of the child to remove
110     */
111    void removeChild(Object id);
112
113    /**
114     * Get the parent {@link TreeNode}.
115     * 
116     * @return parent {@link TreeNode} instance or <code>null</code> if the node is the root
117     */
118    TreeNode<T> getParent();
119
120    /**
121     * Set the parent {@link TreeNode}.
122     * 
123     * @param parent
124     *            {@link TreeNode} to set as parent
125     */
126    void setParent(TreeNode<T> parent);
127}
