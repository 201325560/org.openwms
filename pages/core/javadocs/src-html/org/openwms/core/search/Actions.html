022package org.openwms.core.search;
023
024import javax.xml.bind.annotation.XmlAccessType;
025import javax.xml.bind.annotation.XmlAccessorType;
026import javax.xml.bind.annotation.XmlAttribute;
027import javax.xml.bind.annotation.XmlType;
028import java.io.Serializable;
029import java.util.ArrayList;
030import java.util.List;
031
032/**
033 * Encapsulates a collection of {@link Action}s and is mapped to the XML type "actions" in the ui-actions-schema.xsd. <p> <a
034 * href="http://www.openwms.org/schema/ui-actions-schema.xsd">http://www. openwms.org/schema/ui-actions-schema.xsd</a> </p>
035 *
036 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
037 * @version $Revision$
038 * @since 0.2
039 */
040@XmlAccessorType(XmlAccessType.FIELD)
041@XmlType(name = "actions", propOrder = {"action"})
042public class Actions implements Serializable {
043
044    private static final long serialVersionUID = -8134374894573948880L;
045    /**
046     * A List of all {@link Action}s.
047     */
048    protected List<Action> action;
049    /**
050     * The owning module of the set of {@link Action}s.
051     */
052    @XmlAttribute
053    protected String owner;
054
055    /**
056     * Gets the value of the action property.
057     * <p>
058     * <p>
059     * This acessor method returns a reference to the live list, not a snapshot. Therefore any modification you make to the returned list
060     * will be present inside the JAXB object. This is why there is not a <CODE>set</CODE> method for the action property.
061     * <p>
062     * <p>
063     * For example, to add a new item, do as follows:
064     * <p>
065     * <pre>
066     * getAction().add(newItem);
067     * </pre>
068     * <p>
069     * <p>
070     * <p>
071     * Objects of the following type(s) are allowed in the list {@link Action }
072     *
073     * @return The list of {@link Action}s
074     */
075    public List<Action> getAction() {
076        if (action == null) {
077            action = new ArrayList<Action>();
078        }
079        return action;
080    }
081
082    /**
083     * Gets the owner of this property.
084     *
085     * @return possible object is {@link String }
086     */
087    public String getOwner() {
088        return owner;
089    }
090
091    /**
092     * Sets the owner of this property.
093     *
094     * @param owner allowed object is {@link String }
095     */
096    public void setOwner(String owner) {
097        this.owner = owner;
098    }
099}
