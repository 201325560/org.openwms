022package org.openwms.core;
023
024import javax.persistence.Column;
025import javax.persistence.EntityListeners;
026import javax.persistence.MappedSuperclass;
027import javax.persistence.PrePersist;
028import javax.persistence.PreUpdate;
029import javax.persistence.Temporal;
030import javax.persistence.TemporalType;
031import javax.xml.bind.annotation.XmlTransient;
032import java.io.Serializable;
033import java.util.Date;
034import java.util.UUID;
035
036/**
037 * An AbstractEntity, used as a base class for all domain classes.
038 * <p>
039 * Adds an unique identified to a subclassed domain class that is created when the class is instantiated on the client tier. At least this
040 * uid is created before the instance is persisted the first time. The uid may not be mistaken with the id property that is usually used for
041 * database identity (primary key) or with a business key column.
042 * </p>
043 * <p>
044 * This class has an inner static declared class that is registered as a JPA EntityListener and forces the creation of an uid if not already
045 * created before. This assures that each persisted entity has an uid.
046 * </p>
047 * <p>
048 * The uid property is used by the ActionScript client application to synchronize client-side entity instances with server-side ones.
049 * </p>
050 * <strong>NOTE:</strong><br />
051 * This class uses the uid for comparison with {@link #equals(Object)} and calculation of {@link #hashCode()}.
052 *
053 * @param <ID> The type of technical key
054 * @author <a href="mailto:russelltina@users.sourceforge.net">Tina Russell</a>
055 * @version $Revision$
056 * @since 0.1
057 */
058@XmlTransient
059@MappedSuperclass
060@EntityListeners({ AbstractEntity.AbstractEntityListener.class })
061public abstract class AbstractEntity<ID extends Serializable> implements DomainObject<ID> {
062
063    private static final int UID_LENGTH = 36;
064    /**
065     * Suffix for the FIND_ALL named query. Default {@value}
066     */
067    public static final String FIND_ALL = ".findAll";
068
069    /**
070     * Unique identifier column, used for ActionScript clients.
071     */
072    /* "UUID" and "UID" are Oracle reserved keywords -> "ENTITY_UID" */
073    @XmlTransient
074    @Column(name = "C_ENTITY_UID", unique = true, nullable = false, updatable = false, length = UID_LENGTH)
075    private String uid;
076
077    /** The date this product has been created. */
078    @Temporal(TemporalType.TIMESTAMP)
079    @Column(name = "C_CREATED_DT")
080    private Date createdDate;
081
082    /** The date this LoadUnit has changed the last time. */
083    @Temporal(TemporalType.TIMESTAMP)
084    @Column(name = "C_CHANGED_DT")
085    private Date changedDate;
086
087
088
089    /**
090     * {@inheritDoc}
091     * 
092     * Compare the uid property field.
093     * 
094     * @see java.lang.Object#equals(Object)
095     */
096    @Override
097    public boolean equals(Object o) {
098        return (o == this || (o instanceof AbstractEntity && uid().equals(((AbstractEntity) o).uid())));
099    }
100
101    /**
102     * {@inheritDoc}
103     * 
104     * Use the uid to calculate the hashCode.
105     * 
106     * @see java.lang.String#hashCode()
107     */
108    @Override
109    public int hashCode() {
110        return uid().hashCode();
111    }
112
113    /**
114     * An AbstractEntityListener forces the creation of an uid before the entity is persisted.
115     * 
116     * @author <a href="mailto:russelltina@users.sourceforge.net">Tina Russell</a>
117     * @version $Revision$
118     * @since 0.1
119     */
120    @XmlTransient
121    public static class AbstractEntityListener {
122
123        /**
124         * Before a new entity is persisted we generate an UUID for it.
125         * 
126         * @param abstractEntity
127         *            The entity to persist
128         */
129        @PrePersist
130        public void onPreInsert(AbstractEntity abstractEntity) {
131            abstractEntity.uid();
132        }
133    }
134
135    /**
136     * Set the creation date.
137     */
138    @PrePersist
139    void prePersist() {
140        this.createdDate = new Date();
141    }
142
143    /**
144     * Set the changed date.
145     */
146    @PreUpdate
147    void preUpdate() {
148        this.changedDate = new Date();
149    }
150
151    private String uid() {
152        if (uid == null) {
153            uid = UUID.randomUUID().toString();
154        }
155        return uid;
156    }
157}
