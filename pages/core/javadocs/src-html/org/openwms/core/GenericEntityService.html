022package org.openwms.core;
023
024import java.io.Serializable;
025import java.util.Collection;
026import java.util.List;
027
028/**
029 * A GenericEntityService provides some common used CRUD operations available for most entity types.
030 *
031 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
032 * @version $Revision: $
033 * @since 0.2
034 */
035public interface GenericEntityService<T extends AbstractEntity, ID extends Serializable, BK extends Serializable> {
036
037    /**
038     * Create and persist a new entity.
039     *
040     * @param entity The new entity to create
041     * @return The newly created and persisted <tt>entity</tt>
042     */
043    T create(T entity);
044
045    /**
046     * Return an entity identified by it's unique business key.
047     *
048     * @param key The business key to identify the entity
049     * @return The entity
050     */
051    T findByBK(BK key);
052
053    /**
054     * Find and return all entities.
055     *
056     * @return A list of all entities
057     */
058    List<T> findAll();
059
060    /**
061     * Find and return an entity identified by it's unique technical <tt>id</tt> .
062     *
063     * @param id The unique technical identifier of the entity
064     * @return The entity
065     */
066    T findById(ID id);
067
068    /**
069     * Save the given entity or persist it, in case it is a transient one.
070     *
071     * @param entity Entity to save
072     * @return Saved entity instance
073     */
074    T save(T entity);
075
076    /**
077     * Save an collection of entities. No matter if they already exist or not.
078     *
079     * @param entities An collection of entities to save
080     * @return A collection of all passed entities containing updated ones as well as not updated ones.
081     */
082    Collection<T> saveAll(Collection<T> entities);
083
084    /**
085     * Remove an <tt>entity</tt>.
086     *
087     * @param entity The entity to remove
088     */
089    void remove(T entity);
090
091    /**
092     * Remove entities identified by their unique business key.
093     *
094     * @param keys The unique business keys of the entities to remove
095     */
096    void removeByBK(BK[] keys);
097
098    /**
099     * Remove entities identified by their unique technical key.
100     *
101     * @param keys The unique technical keys of the entities to remove
102     */
103    void removeByID(ID[] keys);
104}
