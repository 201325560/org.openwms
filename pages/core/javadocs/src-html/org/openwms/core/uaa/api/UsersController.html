050/**
051 * An UsersController represents a RESTful access to <tt>User</tt>s. It is transactional by the means it is the outer application service

069    /**
070     * This method returns all existing <tt>User</tt>s. <p> <p> <table> <tr> <td>URI</td> <td>/users</td> </tr> <tr> <td>Verb</td>
071     * <td>GET</td> </tr> <tr> <td>Auth</td> <td>YES</td> </tr> <tr> <td>Header</td> <td></td> </tr> </table> </p> <p> The response stores
072     * <tt>User</tt> instances JSON encoded. It contains a collection of <tt>User</tt> objects. </p>
073     *
074     * @return JSON response
075     */
076    @RequestMapping(method = RequestMethod.GET)
077    @ResponseBody
078    public ResponseEntity<Response<UserVO>> findAllUsers() {
079        List<UserVO> users = m.map(new ArrayList<>(service.findAll()), UserVO.class);
080        return buildOKResponse(users.toArray(new UserVO[users.size()]));
081    }
082
083    /**
084     * Takes a newly created <tt>User</tt> instance and persists it. <p> <p> <table> <tr> <td>URI</td> <td>/users</td> </tr> <tr>
085     * <td>Verb</td> <td>POST</td> </tr> <tr> <td>Auth</td> <td>YES</td> </tr> <tr> <td>Header</td> <td></td> </tr> </table> </p> <p>
086     * Request Body <p>
087     * <pre>
088     *   {
089     *     "username" : "testuser"
090     *   }
091     * </pre>
092     * <p> Parameters: <ul> <li>username (String):</li> The unique username. </ul> </p> <p> Response Body <p>
093     * <pre>
094     *   {
095     *     "id" : 4711,
096     *     "username" : "testuser",
097     *     "token" : "876sjh36ejwhd",
098     *     "version" : 1
099     *   }
100     * </pre>
101     * <p> <ul> <li>id (Integer (32bit)):</li> The internal unique technical key for the stored instance. <li>username (String):</li> The
102     * unique username. <li>token (String):</li> A generated token that is used to authenticate each request. <li>version (Integer
103     * (32bit)):</li> A version number used internally for optimistic locking. </ul> </p>
104     *
105     * @param user The user to create
106     * @return a responseVO
107     */
108    @RequestMapping(method = RequestMethod.POST)
109    @ResponseBody
110    public ResponseEntity<Response<UserVO>> create(@RequestBody @Valid @NotNull UserVO user, HttpServletRequest req, HttpServletResponse resp) {
111        User createdUser = service.create(m.map(user, User.class));
112        resp.addHeader(HttpHeaders.LOCATION, getLocationForCreatedResource(req, createdUser.getPk().toString()));
113        return buildResponse(HttpStatus.CREATED, translate(Messages.CREATED), Messages.CREATED);
114    }
115
116    /**
117     * FIXME [scherrer] Comment this

119     * @param user
120     * @return a responseVO
121     */
122    @RequestMapping(method = RequestMethod.PUT)
123    @ResponseBody
124    public ResponseEntity<Response<UserVO>> save(@RequestBody @Valid UserVO user) {
125        User eo = m.map(user, User.class);
126        UserVO saved = m.map(service.save(eo), UserVO.class);
127
128        if (user.getId() == null) {
129            return buildResponse(HttpStatus.CREATED, translate(Messages.CREATED), Messages.CREATED, saved);
130        } else {
131            return buildResponse(HttpStatus.OK, translate(Messages.SERVER_OK), Messages.SERVER_OK, saved);
132        }
133    }

183    private User findByUsername(String pUsername) {
184        Collection<User> users = service.findAll();
185        for (User user : users) {
186            if (user.getUsername().equals(pUsername)) {
187                return user;
188            }
189        }
190        return null;
191    }
