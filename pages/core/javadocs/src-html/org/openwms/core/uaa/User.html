055/**
056 * An User represents a human user of the system. Typically an User is assigned to one or more {@code Roles} to define security constraints.
057 * Users can have their own configuration settings in form of {@code UserPreferences} and certain user details, encapsulated in an {@code
058 * UserDetails} object that tend to be extended by projects.
059 *
060 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>

063 * @see UserDetails
064 * @see UserPassword
065 * @see org.openwms.core.uaa.Role
066 * @since 0.1
067 */

261    /**
262     * Checks if the new password is a valid and change the password of this User.
263     *
264     * @param encodedPassword The new encoded password of this User
265     * @throws InvalidPasswordException in case changing the password is not allowed or the new password is not valid
266     */
267    public void changePassword(String encodedPassword, String rawPassword, PasswordEncoder encoder) throws InvalidPasswordException {
268        if (persistedPassword != null && encoder.matches(rawPassword ,persistedPassword)) {
269            LOGGER.debug("Trying to set the new password equals to the current password");
270            return;
271        }
272        validateAgainstPasswordHistory(rawPassword, encoder);
273        storeOldPassword(this.password);
274        persistedPassword = encodedPassword;
275        this.password = encodedPassword;
276        lastPasswordChange = new Date();
277    }
278
279    /**
280     * Checks whether the password is going to change.
281     *
282     * @return {@literal true} when {@code password} is different to the originally persisted one, otherwise {@literal false}
283     */
284    public boolean hasPasswordChanged() {
285        return (persistedPassword.equals(password));
286    }
287
288    /**
289     * Check whether the new password is in the history of former passwords.
290     *
291     * @param rawPassword The password to verify
292     * @return {@literal true} if the password is valid, otherwise {@literal false}
293     */
294    protected void validateAgainstPasswordHistory(String rawPassword, PasswordEncoder encoder) throws InvalidPasswordException {
295        for(UserPassword up : passwords) {
296            if (encoder.matches(rawPassword, up.getPassword())) {
297                throw new InvalidPasswordException("Password is not confirm with defined rules");
298            }
299        }
300    }
301
302    private void storeOldPassword(String oldPassword) {
303        if (oldPassword == null || oldPassword.isEmpty()) {
304            if (LOGGER.isDebugEnabled()) {
305                LOGGER.debug("If the old password is null, do not store it in history");
306            }
307            return;
308        }
309        passwords.add(new UserPassword(this, oldPassword));
310        if (passwords.size() > NUMBER_STORED_PASSWORDS) {
311            Collections.sort(passwords, new PasswordComparator());
312            if (LOGGER.isDebugEnabled()) {
313                LOGGER.debug("Remove the old password from the history: " + passwords.get(passwords.size() - 1));
314            }
315            UserPassword pw = passwords.get(passwords.size() - 1);
316            pw.setUser(null);
317            passwords.remove(pw);
318        }
319    }

455    /**
456     * {@inheritDoc}
457     * <p>
458     * Does not call the superclass. Uses the username for calculation.
459     *
460     * @see java.lang.Object#hashCode()
461     */
462    @Override
463    public int hashCode() {
464        final int prime = 31;
465        int result = 1;
466        result = prime * result + ((username == null) ? 0 : username.hashCode());
467        return result;
468    }
469
470    /**
471     * {@inheritDoc}
472     * <p>
473     * Uses the username for comparison.
474     *
475     * @see java.lang.Object#equals(java.lang.Object)
476     */
477    @Override
478    public boolean equals(Object obj) {
479        if (this == obj) {
480            return true;
481        }
482        if (!(obj instanceof User)) {
483            return false;
484        }
485        User other = (User) obj;
486        if (username == null) {
487            if (other.username != null) {
488                return false;
489            }
490        } else if (!username.equals(other.username)) {
491            return false;
492        }
493        return true;
494    }

506    /**
507     * A PasswordComparator sorts UserPassword by date ascending.
508     *
509     * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
510     * @version $Revision$
511     * @since 0.2
512     */
513    static class PasswordComparator implements Comparator<UserPassword>, Serializable {
514
515        /**
516         * {@inheritDoc}
517         *
518         * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
519         */
520        @Override
521        public int compare(UserPassword o1, UserPassword o2) {
522            return o2.getPasswordChanged().compareTo(o1.getPasswordChanged());
523        }
524    }
