022package org.openwms.core.uaa;
023
024import javax.persistence.Column;
025import javax.persistence.DiscriminatorColumn;
026import javax.persistence.DiscriminatorType;
027import javax.persistence.Entity;
028import javax.persistence.Inheritance;
029import javax.persistence.OrderBy;
030import javax.persistence.Table;
031import java.io.Serializable;
032
033import org.ameba.integration.jpa.BaseEntity;
034import org.openwms.core.values.CoreTypeDefinitions;
035import org.springframework.util.Assert;
036
037/**
038 * A SecurityObject is the generalization of {@code Role}s and {@code Grant}s and combines common used properties of both.
039 *
040 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
041 * @version 0.2
042 * @GlossaryTerm
043 * @see org.openwms.core.uaa.Role
044 * @see org.openwms.core.uaa.Grant
045 * @since 0.1
046 */
047@Entity
048@Table(name = "COR_ROLE")
049@Inheritance
050@DiscriminatorColumn(name = "TYPE", discriminatorType = DiscriminatorType.STRING, length = 20)
051public class SecurityObject extends BaseEntity implements Serializable {
052
053    /** Unique name of the {@code SecurityObject}. */
054    @Column(name = "C_NAME", unique = true)
055    @OrderBy
056    private String name;
057
058    /** Description of the {@code SecurityObject}. */
059    @Column(name = "C_DESCRIPTION", length = CoreTypeDefinitions.DESCRIPTION_LENGTH)
060    private String description;
061
062    /* ----------------------------- methods ------------------- */
063
064    /**
065     * Dear JPA...
066     */
067    protected SecurityObject() {
068    }
069
070    /**
071     * Create a new {@code SecurityObject} with a name.
072     *
073     * @param name The name of the {@code SecurityObject}
074     * @throws IllegalArgumentException if name is {@literal null} or an empty String
075     */
076    public SecurityObject(String name) {
077        Assert.hasText(name, "A name of a SecurityObject must not be null");
078        this.name = name;
079    }
080
081    /**
082     * Create a new {@code SecurityObject} with name and description.
083     *
084     * @param name The name of the {@code SecurityObject}
085     * @param description The description text of the {@code SecurityObject}
086     * @throws IllegalArgumentException if name is {@literal null} or an empty String
087     */
088    public SecurityObject(String name, String description) {
089        Assert.hasText(name, "A name of a SecurityObject must not be null");
090        this.name = name;
091        this.description = description;
092    }
093
094    /**
095     * Returns the name.
096     *
097     * @return The name of the {@code SecurityObject}
098     */
099    public String getName() {
100        return name;
101    }
102
103    /**
104     * Returns the description text.
105     *
106     * @return The description of the {@code SecurityObject} as text
107     */
108    public String getDescription() {
109        return description;
110    }
111
112    /**
113     * Set the description for the {@code SecurityObject}.
114     *
115     * @param description The description of the {@code SecurityObject} as text
116     */
117    public void setDescription(String description) {
118        this.description = description;
119    }
120
121    /**
122     * {@inheritDoc}
123     *
124     * @see java.lang.Object#hashCode()
125     */
126    @Override
127    public int hashCode() {
128        final int prime = 31;
129        int result = 1;
130        result = prime * result + ((name == null) ? 0 : name.hashCode());
131        return result;
132    }
133
134    /**
135     * {@inheritDoc} Compare the name.
136     *
137     * @see java.lang.Object#equals(java.lang.Object)
138     */
139    @Override
140    public boolean equals(Object obj) {
141        if (null == obj) {
142            return false;
143        }
144        if (this == obj) {
145            return true;
146        }
147        if (getClass() != obj.getClass()) {
148            return false;
149        }
150        SecurityObject other = (SecurityObject) obj;
151        if (name == null) {
152            if (other.name != null) {
153                return false;
154            }
155        } else if (!name.equals(other.name)) {
156            return false;
157        }
158        return true;
159    }
160
161    /**
162     * Return the name.
163     *
164     * @return the name
165     */
166    @Override
167    public String toString() {
168        return name;
169    }
170}
