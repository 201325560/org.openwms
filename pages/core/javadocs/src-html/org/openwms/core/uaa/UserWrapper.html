022package org.openwms.core.uaa;
023
024import java.util.Collection;
025import java.util.HashSet;
026import java.util.stream.Collectors;
027
028import org.springframework.security.core.GrantedAuthority;
029import org.springframework.security.core.userdetails.UserDetails;
030import org.springframework.util.Assert;
031
032/**
033 * An UserWrapper is used as an adapter between {@link Role}s, {@link SecurityObject}s and Spring's {@link GrantedAuthority} objects.
034 *
035 * @author <a href="mailto:russelltina@users.sourceforge.net">Tina Russell</a>
036 * @version 0.2
037 * @see org.openwms.core.uaa.SecurityObject
038 * @see org.openwms.core.uaa.Role
039 * @see org.springframework.security.core.GrantedAuthority
040 * @see org.springframework.security.core.userdetails.UserDetails
041 * @since 0.1
042 */
043public class UserWrapper implements UserDetails, UserHolder {
044
045    private User user;
046    private Collection<GrantedAuthority> authorities = null;
047
048    /**
049     * Create a new UserWrapper.
050     *
051     * @param user The User to wrap
052     */
053    public UserWrapper(User user) {
054        Assert.notNull(user, "Not allowed to create an UserWrapper with null argument");
055        this.user = user;
056        this.user.getRoles().size();
057    }
058
059    /**
060     * Subclasses can set a collection of grants that are always available for an User. This is useful for administrative accounts.
061     *
062     * @param authz A collection of grants (authorities) where the default grants are added to
063     */
064    protected void addDefaultGrants(Collection<GrantedAuthority> authz) {
065    }
066
067    /**
068     * {@inheritDoc}
069     */
070    @Override
071    public User getUser() {
072        return user;
073    }
074
075    /**
076     * {@inheritDoc}
077     *
078     * @return the authorities, sorted by natural key (never {@literal null})
079     */
080    @Override
081    public Collection<GrantedAuthority> getAuthorities() {
082        if (null == authorities) {
083            authorities = new HashSet<>();
084            authorities.addAll(user.getGrants().stream().map(SecurityObjectAuthority::new).collect(Collectors.toList()));
085            addDefaultGrants(authorities);
086        }
087        return authorities;
088    }
089
090    /**
091     * {@inheritDoc}
092     *
093     * @return the password (never {@literal null})
094     */
095    @Override
096    public String getPassword() {
097        return user.getPassword();
098    }
099
100    /**
101     * {@inheritDoc}
102     *
103     * @return the username (never {@literal null})
104     */
105    @Override
106    public String getUsername() {
107        return user.getUsername();
108    }
109
110    /**
111     * {@inheritDoc}
112     *
113     * @return {@literal true} if the user's account is valid (ie non-expired), {@literal false} if no longer valid (ie expired)
114     */
115    @Override
116    public boolean isAccountNonExpired() {
117        return true;
118    }
119
120    /**
121     * {@inheritDoc}
122     *
123     * @return {@literal true} if the user is not locked, {@literal false} otherwise
124     */
125    @Override
126    public boolean isAccountNonLocked() {
127        return !user.isLocked();
128    }
129
130    /**
131     * {@inheritDoc}
132     *
133     * @return {@literal true} if the user's credentials are valid (ie non-expired), {@literal false} if no longer valid (ie expired)
134     * @see org.springframework.security.core.userdetails.UserDetails#isCredentialsNonExpired()
135     */
136    @Override
137    public boolean isCredentialsNonExpired() {
138        return true;
139    }
140
141    /**
142     * {@inheritDoc}
143     *
144     * @return {@literal true} if the user is enabled, {@literal false} otherwise
145     */
146    @Override
147    public boolean isEnabled() {
148        return user.isEnabled();
149    }
150
151    /**
152     * {@inheritDoc}
153     * <p>
154     * Uses authorities and user for calculation.
155     */
156    @Override
157    public int hashCode() {
158        final int prime = 31;
159        int result = 1;
160        result = prime * result + ((authorities == null) ? 0 : authorities.hashCode());
161        result = prime * result + ((user == null) ? 0 : user.hashCode());
162        return result;
163    }
164
165    /**
166     * {@inheritDoc}
167     * <p>
168     * Uses authorities and user for comparison.
169     */
170    @Override
171    public boolean equals(Object obj) {
172        if (this == obj) {
173            return true;
174        }
175        if (obj == null) {
176            return false;
177        }
178        if (getClass() != obj.getClass()) {
179            return false;
180        }
181        UserWrapper other = (UserWrapper) obj;
182        if (authorities == null) {
183            if (other.authorities != null) {
184                return false;
185            }
186        } else if (!authorities.equals(other.authorities)) {
187            return false;
188        }
189        if (user == null) {
190            if (other.user != null) {
191                return false;
192            }
193        } else if (!user.equals(other.user)) {
194            return false;
195        }
196        return true;
197    }
198
199    /**
200     * Return the Users username.
201     *
202     * @return The username of the User
203     * @see java.lang.Object#toString()
204     */
205    @Override
206    public String toString() {
207        return getUsername();
208    }
209}
