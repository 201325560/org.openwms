022package org.openwms.core;
023
024import javax.persistence.EntityManager;
025import javax.persistence.PersistenceContext;
026import javax.persistence.Query;
027import java.io.Serializable;
028import java.util.Collections;
029import java.util.List;
030import java.util.Map;
031
032import org.openwms.core.exception.ExceptionCodes;
033import org.openwms.core.exception.NoUniqueResultException;
034import org.springframework.beans.factory.annotation.Autowired;
035import org.springframework.context.MessageSource;
036import org.springframework.transaction.annotation.Propagation;
037import org.springframework.transaction.annotation.Transactional;
038
039/**
040 * An AbstractGenericJpaDao - Extend this DAO implementation to inherit simple JPA CRUD operations.
041 * <p>
042 * This {@link GenericDao} implementation acts as a transactional superclass to have a benefit from Springs exception translation and
043 * transaction management.
044 * </p>
045 * <p>
046 * Furthermore an {@link AbstractGenericJpaDao} has transactional behavior expressed with Springs {@link Transactional} annotation.
047 * </p>
048 * 
049 * @param <T>
050 *            Any serializable type, mostly an Entity class type.
051 * @param <ID>
052 *            The type of the Entity class' unique id
053 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
054 * @version $Revision$
055 * @since 0.1
056 * @see org.springframework.stereotype.Repository
057 * @see org.springframework.transaction.annotation.Transactional
058 */
059@Transactional(propagation = Propagation.MANDATORY)
060public abstract class AbstractGenericJpaDao<T extends AbstractEntity<ID>, ID extends Serializable> implements
061        GenericDao<T, ID> {
062
063    @PersistenceContext
064    private EntityManager em;
065    @Autowired
066    private MessageSource messageSource;
067
068    /**
069     * Create a new AbstractGenericJpaDao.
070     */
071    protected AbstractGenericJpaDao() {
072
073    }
074
075    /**
076     * Returns the entity class to deals with.<br>
077     * The Java Reflection API is used to find the type.
078     * 
079     * @return Entity class type.
080     */
081    protected abstract Class<T> getPersistentClass();
082
083    /**
084     * {@inheritDoc}
085     */
086    @Override
087    public T findById(ID id) {
088        return em.find(getPersistentClass(), id);
089    }
090
091    /**
092     * {@inheritDoc}
093     * 
094     * This implementation never return <code>null</code>. In case of no Entities were found an empty List is returned.
095     */
096    @SuppressWarnings("unchecked")
097    @Override
098    public List<T> findAll() {
099        List<T> all = em.createNamedQuery(getFindAllQuery()).getResultList();
100        return all == null ? Collections.<T> emptyList() : all;
101    }
102
103    /**
104     * {@inheritDoc}
105     */
106    @Override
107    @SuppressWarnings("unchecked")
108    public List<T> findByNamedParameters(String queryName, Map<String, ?> params) {
109        Query queryObject = em.createNamedQuery(queryName);
110        if (params != null) {
111            for (Map.Entry<String, ?> entry : params.entrySet()) {
112                queryObject.setParameter(entry.getKey(), entry.getValue());
113            }
114        }
115        return queryObject.getResultList();
116    }
117
118    /**
119     * {@inheritDoc}
120     */
121    @Override
122    @SuppressWarnings("unchecked")
123    public List<T> findByPositionalParameters(String queryName, Object... values) {
124        Query queryObject = em.createNamedQuery(queryName);
125        if (values != null) {
126            for (int i = 0; i < values.length; i++) {
127                queryObject.setParameter(i + 1, values[i]);
128            }
129        }
130        return queryObject.getResultList();
131    }
132
133    /**
134     * {@inheritDoc}
135     * 
136     * A {@link NoUniqueResultException} is thrown when more than one Entity were found. In case of no result, <code>null</code> is
137     * returned.
138     * 
139     * @throws NoUniqueResultException
140     *             when more than one Entity were found
141     */
142    @Override
143    @SuppressWarnings("unchecked")
144    public T findByUniqueId(Serializable id) {
145        List<T> result = em.createNamedQuery(getFindByUniqueIdQuery()).setParameter(1, id).getResultList();
146        if (result.size() > 1) {
147            throw new NoUniqueResultException(messageSource.getMessage(ExceptionCodes.MULIPLE_ENTITIES_FOUND,
148                    new Serializable[] { id }, null));
149        }
150        return result.size() == 0 ? null : result.get(0);
151    }
152
153    /**
154     * {@inheritDoc}
155     */
156    @Override
157    public T save(T entity) {
158        beforeUpdate(entity);
159        return em.merge(entity);
160    }
161
162    /**
163     * {@inheritDoc}
164     */
165    @Override
166    public void remove(T entity) {
167        em.remove(entity);
168    }
169
170    /**
171     * {@inheritDoc}
172     */
173    @Override
174    public void persist(T entity) {
175        beforeUpdate(entity);
176        em.persist(entity);
177    }
178
179    /**
180     * Returns the name of the <code>NamedQuery</code> to find all Entity classes.
181     * 
182     * @return Name of the query
183     */
184    protected abstract String getFindAllQuery();
185
186    /**
187     * Returns the name of the <code>NamedQuery</code> to find an Entity by the business key.
188     * 
189     * @return Name of the query
190     */
191    protected abstract String getFindByUniqueIdQuery();
192
193    /**
194     * This method is considered as a hook to do something before an update is performed.
195     * 
196     * @param entity
197     *            The Entity that is updated
198     */
199    protected void beforeUpdate(T entity) {
200
201    }
202
203    /**
204     * Subclasses can call this method to retrieve an shared {@link EntityManager} instance.
205     * 
206     * @return The {@link EntityManager}
207     */
208    protected final EntityManager getEm() {
209        return em;
210    }
211}
