022package org.openwms.core;
023
024import javax.persistence.PersistenceException;
025import java.io.Serializable;
026import java.util.ArrayList;
027import java.util.Collection;
028import java.util.Collections;
029import java.util.List;
030
031import org.ameba.exception.IntegrationLayerException;
032import org.ameba.exception.NotFoundException;
033import org.ameba.exception.ServiceLayerException;
034import org.openwms.core.exception.ExceptionCodes;
035import org.springframework.beans.factory.annotation.Autowired;
036import org.springframework.context.MessageSource;
037import org.springframework.stereotype.Service;
038import org.springframework.transaction.annotation.Transactional;
039import org.springframework.util.Assert;
040
041/**
042 * A AbstractGenericEntityService.
043 *
044 * @param <BK> The type of business key
045 * @param <ID> The type of technical key
046 * @param <T> The type of Entity
047 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
048 * @version $Revision: $
049 * @since 0.2
050 */
051@Transactional
052@Service
053public abstract class AbstractGenericEntityService<T extends AbstractEntity<ID>, ID extends Serializable, BK extends Serializable>
054        implements GenericEntityService<T, ID, BK> {
055
056    @Autowired
057    private MessageSource messageSource;
058
059    /**
060     * Return a type-safe DAO implementation instantiated by the sub-classing service.
061     *
062     * @return the DAO
063     */
064    protected abstract GenericDao<T, ID> getRepository();
065
066    /**
067     * Get the messageSource.
068     *
069     * @return the messageSource.
070     */
071    protected MessageSource getMessageSource() {
072        return messageSource;
073    }
074
075    /**
076     * Tries to resolve a given <tt>entity</tt> by it's business key and returns it to the caller. In case the <tt>entity</tt> could not be
077     * resolved, <code>null</code> may be returned.
078     */
079    protected abstract T resolveByBK(T entity);
080
081    /**
082     * {@inheritDoc}
083     *
084     * @throws ServiceLayerException if <ul> <li><tt>entity</tt> is <code>null</code></li> <li><tt>entity</tt> does not exist</li> </ul>
085     */
086    @Override
087    public T create(T entity) {
088        checkForNull(entity, ExceptionCodes.ENTITY_NOT_BE_NULL);
089        if (!entity.isNew()) {
090            String msg = getMessageSource().getMessage(ExceptionCodes.ENTITY_ALREADY_EXISTS, new Object[]{entity},
091                    null);
092            throw new ServiceLayerException(msg);
093        }
094        T persistedEntity = resolveByBK(entity);
095        if (persistedEntity != null) {
096            String msg = getMessageSource().getMessage(ExceptionCodes.ENTITY_ALREADY_EXISTS, new Object[]{entity},
097                    null);
098            throw new ServiceLayerException(msg);
099        }
100        getRepository().persist(entity);
101        return getRepository().save(entity);
102    }
103
104    /**
105     * {@inheritDoc}
106     *
107     * @throws NotFoundException if entity with <tt>id</tt> does not exist
108     */
109    @Override
110    public T findById(ID id) {
111        T entity = getRepository().findById(id);
112        if (entity == null) {
113            throw new NotFoundException(getMessageSource().getMessage(ExceptionCodes.ENTITY_NOT_EXIST,
114                    new Object[]{id}, null), ExceptionCodes.ENTITY_NOT_EXIST);
115        }
116        return entity;
117    }
118
119    /**
120     * {@inheritDoc}
121     * <p>
122     * Implementation returns an empty list in case of no result.
123     * <p>
124     * Marked as <code>readOnly</code> transactional method.
125     */
126    @Override
127    @Transactional(readOnly = true)
128    public List<T> findAll() {
129        List<T> result = getRepository().findAll();
130        return result == null ? Collections.<T>emptyList() : result;
131    }
132
133    /**
134     * {@inheritDoc}
135     *
136     * @throws NotFoundException if entity with <code>key</code> does not exist
137     */
138    @Override
139    public T findByBK(BK key) {
140        T role = getRepository().findByUniqueId(key);
141        if (role == null) {
142            throw new NotFoundException(getMessageSource().getMessage(ExceptionCodes.ENTITY_NOT_EXIST,
143                    new Object[]{key}, null), ExceptionCodes.ENTITY_NOT_EXIST);
144        }
145        return role;
146    }
147
148    /**
149     * {@inheritDoc}
150     * <p>
151     * Handles detached and managed entities. Transient instances will be ignored.
152     */
153    @Override
154    public void remove(T entity) {
155        if (!entity.isNew()) {
156
157            // Get and remove the managed instance, to be save
158            T persistedEntity = getRepository().findById(entity.getId());
159            getRepository().remove(persistedEntity);
160        }
161    }
162
163    /**
164     * {@inheritDoc}
165     *
166     * @throws NotFoundException if one of the entities does not exist
167     */
168    @Override
169    public void removeByBK(BK[] keys) {
170        for (BK key : keys) {
171            if (key != null) {
172                T entity = getRepository().findByUniqueId(key);
173                if (entity == null) {
174                    String msg = getMessageSource().getMessage(ExceptionCodes.ENTITY_NOT_EXIST,
175                            new Object[]{key}, null);
176                    throw new NotFoundException(msg, ExceptionCodes.ENTITY_NOT_EXIST);
177                }
178                getRepository().remove(entity);
179            }
180        }
181    }
182
183    /**
184     * {@inheritDoc}
185     *
186     * @throws NotFoundException if one of the entities does not exist
187     */
188    @Override
189    public void removeByID(ID[] keys) {
190        for (ID key : keys) {
191            if (key != null) {
192                T entity = getRepository().findById(key);
193                if (entity == null) {
194                    String msg = getMessageSource().getMessage(ExceptionCodes.ENTITY_NOT_EXIST,
195                            new Object[]{key}, null);
196                    throw new NotFoundException(msg, ExceptionCodes.ENTITY_NOT_EXIST);
197                }
198                getRepository().remove(entity);
199            }
200        }
201    }
202
203    /**
204     * {@inheritDoc}
205     */
206    @Override
207    public T save(T entity) {
208        if (entity.isNew()) {
209            try {
210                getRepository().persist(entity);
211            } catch (PersistenceException | IntegrationLayerException ex) {
212                String msg = getMessageSource().getMessage(ExceptionCodes.ENTITY_ALREADY_EXISTS,
213                        new Object[]{entity}, null);
214                throw new ServiceLayerException(msg);
215            }
216        }
217        return getRepository().save(entity);
218    }
219
220    /**
221     * {@inheritDoc}
222     */
223    @Override
224    public Collection<T> saveAll(Collection<T> entities) {
225        Assert.notEmpty(entities, translate(ExceptionCodes.ENTITY_NOT_BE_NULL));
226        List<T> result = new ArrayList<>(entities.size());
227        entities.forEach(f ->  result.add(f.isNew() ? create(f) : getRepository().save(f)));
228        return result;
229    }
230
231    /**
232     * Throws an ServiceRuntimeException with <tt>msg</tt> as message text when <tt>args</tt> is null.
233     *
234     * @param args The argument to check
235     * @param msg The exception message text
236     */
237    protected void checkForNull(Object args, String msg) {
238
239        if (args == null) {
240            throw new ServiceLayerException(translate(msg));
241        }
242    }
243
244    /**
245     * Translate a given message <tt>code</tt> into a message text with arguments <tt>args</tt>.
246     *
247     * @param code The message code
248     * @param args Interpolated arguments
249     * @return The message text
250     */
251    protected String translate(String code, Object... args) {
252        return messageSource.getMessage(code, args == null ? new Object[0] : args, null);
253    }
254}
